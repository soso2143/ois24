
library(ggplot2)
library(dplyr)


data(mtcars)
mtcars$cyl <- as.factor(mtcars$cyl) # Convertir la colonne 'cyl' en facteur


ggplot(mtcars, aes(x=wt, y=mpg, color=cyl)) +
  geom_point(size=3, alpha=0.7) +  # Ajouter les points
  geom_smooth(method="lm", se=FALSE, linetype="dashed") +  # Ajouter des régressions linéaires
  facet_wrap(~ cyl) +  # Facetter selon le nombre de cylindres
  theme_minimal() +  # Utiliser un thème minimal
  labs(title="Relation entre le poids et la consommation d'essence",
       subtitle="Facetté par le nombre de cylindres",
       x="Poids (1000 lbs)",
       y="Consommation d'essence (mpg)",
       color="Nombre de cylindres") +
  theme(plot.title = element_text(hjust=0.5, face="bold"),
        plot.subtitle = element_text(hjust=0.5))


ggsave("scatter_plot_facet.png", width=10, height=6)
